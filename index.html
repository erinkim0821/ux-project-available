        let score = 0;


        // Get DOM elements

        const restartButton = document.getElementById('restartButton');


        // Function to display the current question

        function displayQuestion() {

            if (currentQuestionIndex < questions.length) {

                questionCategory.textContent = questions[currentQuestionIndex].category;

                questionText.textContent = questions[currentQuestionIndex].text;

            } else {

                showResult();

            }

        }


        // Function to show the result based on the score

        function showResult() {

            questionSection.classList.add('hidden');

            resultSection.classList.remove('hidden');


            let diagnosis = '';

            if (score >= 10) {

                diagnosis = `🎉 축하합니다! 당신은 UX 디자인 프로젝트 수업에 필요한 기본적인 역량을 잘 갖추고 있는 것으로 보입니다. 수업에 참여하시면 많은 것을 배우고, 흥미로운 프로젝트를 경험할 수 있을 것입니다. 적극적인 참여를 환영합니다! (총 '예' ${score}개)`;

                resultSection.classList.remove('bg-green-50');

                resultSection.classList.add('bg-yellow-50'); // Change background for good result

            } else if (score >= 7) {

                resultSection.classList.remove('bg-green-50');

                resultSection.classList.add('bg-blue-50'); // Change background for moderate result

            } else {

                diagnosis = `💡 이 수업은 실무 프로젝트와 유사한 방식으로 진행되며, 일정 수준의 사전 지식과 툴 활용 능력이 요구됩니다. 현재 상태로는 수업 내용을 따라가거나 프로젝트를 수행하는 데 큰 어려움을 겪을 가능성이 높습니다. 이번 학기에는 UX/UI 디자인 기초 수업 또는 Figma 기초 강의를 먼저 수강하며 기본기를 다진 후, 다음 학기에 본 수업을 수강하시는 것을 고려해보시길 추천합니다. 당신의 열정을 응원합니다! (총 '예' ${score}개)`;

        // Event listener for Yes button

        yesButton.addEventListener('click', () => {

            currentQuestionIndex++;

            displayQuestion();

        // Event listener for No button

        noButton.addEventListener('click', () => {

            currentQuestionIndex++;

        });


        // Event listener for Restart button

        restartButton.addEventListener('click', () => {
